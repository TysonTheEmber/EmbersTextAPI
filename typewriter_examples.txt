# Typewriter Effect Examples
# Copy these into item tooltips, quest descriptions, or custom text components

# BASIC EXAMPLES
# ==============

# Simple typewriter - default 20 chars/sec
<typewriter>This text appears character by character</typewriter>

# Fast typing
<typewriter s=40>Quick typing effect!</typewriter>

# Slow typing for dramatic effect
<typewriter s=10>S...l...o...w typing for suspense...</typewriter>

# With initial delay
<typewriter d=1.0>Waits 1 second before typing</typewriter>
<typewriter s=25 d=0.5>Fast typing after short delay</typewriter>

# Cycling animation (repeats)
<typewriter s=15 c=true>This repeats forever...</typewriter>


# COMBINED WITH OTHER EFFECTS
# ============================

# Rainbow typewriter
<rainbow><typewriter>Colorful typing!</typewriter></rainbow>

# Wave typewriter
<wave><typewriter s=25>Wavy typing effect</typewriter></wave>

# Gradient typewriter
<gradient from=#FF0000 to=#0000FF><typewriter>Gradient typing</typewriter></gradient>

# Glitch + typewriter (dramatic)
<glitch f=1.5><typewriter s=20>Gl1tch3d typ1ng</typewriter></glitch>

# Shadow + typewriter (emphasis)
<shadow x=2 y=2 a=0.5><typewriter s=25>Shadowed typing</typewriter></shadow>

# Multiple effects stacked
<rainbow><wave><typewriter s=20>Rainbow wavy typewriter!</typewriter></wave></rainbow>

# Neon + typewriter (cyberpunk feel)
<neon><typewriter s=30>NEON TYPING EFFECT</typewriter></neon>
